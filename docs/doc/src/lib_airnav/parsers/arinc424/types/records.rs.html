1//! # ARINC 424 Common types in the scope of records when parsing
2//! This module contains the common record parsing types for the ARINC 424 data. These are used in all versions of the parser
3//!
4//! The is_primary_record function is used to determine if a record is a primary record but is also
5//! fairly misplaced imo. Will leave it here for now as it is not the highest of prios to refactor this.
6//! Core prio is stability across revisions.
7
8use crate::parsers::arinc424::types::fields::{BLANK, FieldParseError, ParseableField};
9use std::any::type_name;
10
11#[derive(Debug)]
12pub struct RecordParseError {
13    pub message: String,
14}
15impl From<FieldParseError> for RecordParseError {
16    fn from(error: FieldParseError) -> Self {
17        Self {
18            message: if let (Some(column), Some(column_type)) = (error.column, error.column_type) {
19                format!(
20                    "Record parse error: {} at column {} of type {}",
21                    error.message,
22                    column,
23                    column_type.split("::").last().unwrap_or(&column_type)
24                )
25            } else {
26                format!("Record parse error: {}", error.message)
27            },
28        }
29    }
30}
31#[derive(Debug)]
32pub struct RecordField<'a, T> {
33    pub raw_bytes: &'a [u8],
34    pub value: Option<T>,
35}
36impl<'a, T: ParseableField> RecordField<'a, T> {
37    pub fn from_bytes(
38        input: &'a [u8],
39        column: usize,
40        length: usize,
41    ) -> Result<Self, FieldParseError> {
42        // to make it 1:1 with the spec, let's use 1-indexed columns
43        let value = T::from_bytes(&input[column - 1..column - 1 + length])
44            .map_err(|e| e.with_context(type_name::<T>(), column))?;
45        Ok(Self {
46            raw_bytes: &input[column - 1..column - 1 + length],
47            value,
48        })
49    }
50}
51
52// Layout dispatch helpers
53pub fn is_primary_record(input: &[u8], continuation_column: usize) -> bool {
54    matches!(input[continuation_column - 1], b'0' | b'1' | BLANK)
55}
