8pub(super) struct PreferredRouteRecords;
9impl PreferredRouteRecords {
10    const CONTINUATION_COLUMN: usize = 39;
11    const CONTINUATION_APPLICATION_COLUMN: usize = 40;
12
13    pub fn parse(input: &[u8]) -> Result<ARINCRecord<'_>, RecordParseError> {
14        if is_primary_record(input, Self::CONTINUATION_COLUMN) {
15            Ok(ARINCRecord::PreferredRoutePrimary(
16                PreferredRoutePrimaryRecord::parse(input)?,
17            ))
18        } else {
19            match ContinuationRecordApplicationType::from_bytes(
20                &input[Self::CONTINUATION_APPLICATION_COLUMN - 1
21                    ..Self::CONTINUATION_APPLICATION_COLUMN],
22            )? {
23                Some(ContinuationRecordApplicationType::StandardContinuation) => {
24                    Ok(ARINCRecord::PreferredRouteTimeContinuation(
25                        PreferredRouteTimeContinuationRecord::parse(input)?,
26                    ))
27                }
28                _ => Err(RecordParseError::new(
29                    "Invalid continuation record application type".to_string(),
30                    Some(String::from_utf8_lossy(input).into_owned()),
31                )),
32            }
33        }
34    }
35}
