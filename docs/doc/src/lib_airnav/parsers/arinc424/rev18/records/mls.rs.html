2use crate::parsers::arinc424::rev18::records::record::ARINCRecord;
3use crate::parsers::arinc424::types::fields::ParseableField;
4use crate::parsers::arinc424::types::records::{RecordField, RecordParseError, is_primary_record};
5use crate::parsers::arinc424::rev18::definitions::*;
6pub(super) struct MLSRecords;
7impl MLSRecords {
8    const CONTINUATION_COLUMN: usize = 22;
9    const CONTINUATION_APPLICATION_COLUMN: usize = 23;
10
11    pub fn parse(input: &[u8]) -> Result<ARINCRecord<'_>, RecordParseError> {
12        if is_primary_record(input, Self::CONTINUATION_COLUMN) {
13            Ok(ARINCRecord::MLSPrimary(MLSPrimaryRecord::parse(input)?))
14        } else {
15            match ContinuationRecordApplicationType::from_bytes(
16                &input[Self::CONTINUATION_APPLICATION_COLUMN - 1
17                    ..Self::CONTINUATION_APPLICATION_COLUMN],
18            )? {
19                Some(ContinuationRecordApplicationType::StandardContinuation) => Ok(
20                    ARINCRecord::MLSContinuation(MLSContinuationRecord::parse(input)?),
21                ),
22                _ => Err(RecordParseError {
23                    message: "Invalid continuation record application type".to_string(),
24                }),
25            }
26        }
27    }
28}
