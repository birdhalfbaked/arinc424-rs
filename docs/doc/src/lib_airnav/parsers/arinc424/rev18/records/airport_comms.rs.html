8pub(super) struct AirportCommsRecords;
9impl AirportCommsRecords {
10    const CONTINUATION_COLUMN: usize = 26;
11    const CONTINUATION_APPLICATION_COLUMN: usize = 27;
12
13    pub fn parse(input: &[u8]) -> Result<ARINCRecord<'_>, RecordParseError> {
14        if is_primary_record(input, Self::CONTINUATION_COLUMN) {
15            Ok(ARINCRecord::AirportCommsPrimary(
16                AirportCommsPrimaryRecord::parse(input)?,
17            ))
18        } else {
19            match ContinuationRecordApplicationType::from_bytes(
20                &input[Self::CONTINUATION_APPLICATION_COLUMN
21                    ..Self::CONTINUATION_APPLICATION_COLUMN + 1],
22            )? {
23                Some(ContinuationRecordApplicationType::SectorNarrativeContinuation) => {
24                    Ok(ARINCRecord::AirportCommsSectorNarrativeContinuation(
25                        AirportCommsSectorNarrativeContinuationRecord::parse(input)?,
26                    ))
27                }
28                Some(ContinuationRecordApplicationType::FormattedTimeOfOperationsContinuation) => {
29                    Ok(ARINCRecord::AirportCommsTimeContinuation(
30                        AirportCommsTimeContinuationRecord::parse(input)?,
31                    ))
32                }
33                _ => {
34                    return Err(RecordParseError::new(
35                        "Invalid continuation record application type".to_string(),
36                        Some(String::from_utf8_lossy(input).into_owned()),
37                    ));
38                }
39            }
40        }
41    }
42}

44/// Parse the communications frequency from the input bytes
45/// since we need to manually construct the frequency based on the frequency unit
46fn parse_communications_frequency<'a>(
47    input: &'a [u8],
48) -> Result<RecordField<'a, CommunicationsFrequency>, RecordParseError> {
49    let unit_bytes: &[u8] = &input[24..25];
50    let communications_frequency_bytes: &[u8] = &input[16..23];
51    let communications_frequency =
52        CommunicationsFrequency::parse(unit_bytes, communications_frequency_bytes)?;
53    Ok(RecordField {
54        raw_bytes: communications_frequency_bytes,
55        value: communications_frequency,
56        start_column: 16,
57        end_column: 23,
58    })
59}
