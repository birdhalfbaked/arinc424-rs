23fn parse_pad_dimensions<'a>(
24    input: &'a [u8],
25) -> Result<
26    (
27        RecordField<'a, PadDimensions>,
28        RecordField<'a, PadDimensions>,
29        RecordField<'a, PadDimensions>,
30    ),
31    RecordParseError,
32> {
33    let shape_bytes = &input[22..23];
34    let tlof_dimension_bytes = &input[23..30];
35    let fato_dimension_bytes = &input[70..77];
36    let safety_area_dimension_bytes = &input[78..85];
37
38    let tlof_dimension_value = match HelipadShape::from_bytes(shape_bytes)? {
39        Some(HelipadShape::Circle) => {
40            if let Some(circular_pad_dimensions) =
41                CircularPadDimensions::from_bytes(tlof_dimension_bytes)?
42            {
43                Some(PadDimensions::Circular(circular_pad_dimensions))
44            } else {
45                None
46            }
47        }
48
49        Some(HelipadShape::Rectangular) | Some(HelipadShape::Runway) => {
50            if let Some(rectangular_pad_dimensions) =
51                RectangularPadDimensions::from_bytes(tlof_dimension_bytes)?
52            {
53                Some(PadDimensions::Rectangular(rectangular_pad_dimensions))
54            } else {
55                None
56            }
57        }
58        Some(HelipadShape::Undefined) => None,
59        None => {
60            return Err(RecordParseError {
61                message: "Invalid helipad shape".to_string(),
62            });
63        }
64    };
65    let fato_dimension_value = match HelipadShape::from_bytes(shape_bytes)? {
66        Some(HelipadShape::Circle) => {
67            if let Some(circular_pad_dimensions) =
68                CircularPadDimensions::from_bytes(fato_dimension_bytes)?
69            {
70                Some(PadDimensions::Circular(circular_pad_dimensions))
71            } else {
72                None
73            }
74        }
75
76        Some(HelipadShape::Rectangular) | Some(HelipadShape::Runway) => {
77            if let Some(rectangular_pad_dimensions) =
78                RectangularPadDimensions::from_bytes(fato_dimension_bytes)?
79            {
80                Some(PadDimensions::Rectangular(rectangular_pad_dimensions))
81            } else {
82                None
83            }
84        }
85        Some(HelipadShape::Undefined) => None,
86        None => {
87            return Err(RecordParseError {
88                message: "Invalid helipad shape".to_string(),
89            });
90        }
91    };
92    let safety_area_dimension_value = match HelipadShape::from_bytes(shape_bytes)? {
93        Some(HelipadShape::Circle) => {
94            if let Some(circular_pad_dimensions) =
95                CircularPadDimensions::from_bytes(safety_area_dimension_bytes)?
96            {
97                Some(PadDimensions::Circular(circular_pad_dimensions))
98            } else {
99                None
100            }
101        }
102
103        Some(HelipadShape::Rectangular) | Some(HelipadShape::Runway) => {
104            if let Some(rectangular_pad_dimensions) =
105                RectangularPadDimensions::from_bytes(safety_area_dimension_bytes)?
106            {
107                Some(PadDimensions::Rectangular(rectangular_pad_dimensions))
108            } else {
109                None
110            }
111        }
112        Some(HelipadShape::Undefined) => None,
113        None => {
114            return Err(RecordParseError {
115                message: "Invalid helipad shape".to_string(),
116            });
117        }
118    };
119
120    Ok((
121        RecordField {
122            raw_bytes: tlof_dimension_bytes,
123            value: tlof_dimension_value,
124        },
125        RecordField {
126            raw_bytes: fato_dimension_bytes,
127            value: fato_dimension_value,
128        },
129        RecordField {
130            raw_bytes: safety_area_dimension_bytes,
131            value: safety_area_dimension_value,
132        },
133    ))
134}
