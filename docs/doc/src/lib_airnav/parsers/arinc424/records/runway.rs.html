1use crate::parsers::arinc424::definitions::*;
2use crate::parsers::arinc424::fields::ParseableField;
3use crate::parsers::arinc424::records::record::{
4    ARINCRecord, RecordField, RecordParseError, is_primary_record,
5};
6pub(super) struct RunwayRecords;
7impl RunwayRecords {
8    const CONTINUATION_COLUMN: usize = 22;
9    const CONTINUATION_APPLICATION_COLUMN: usize = 23;
10
11    pub fn parse(input: &[u8]) -> Result<ARINCRecord<'_>, RecordParseError> {
12        if is_primary_record(input, Self::CONTINUATION_COLUMN) {
13            Ok(ARINCRecord::RunwayPrimary(RunwayPrimaryRecord::parse(
14                input,
15            )?))
16        } else {
17            match ContinuationRecordApplicationType::from_bytes(
18                &input[Self::CONTINUATION_APPLICATION_COLUMN - 1
19                    ..Self::CONTINUATION_APPLICATION_COLUMN],
20            )? {
21                Some(ContinuationRecordApplicationType::StandardContinuation) => Ok(
22                    ARINCRecord::RunwayContinuation(RunwayContinuationRecord::parse(input)?),
23                ),
24                Some(ContinuationRecordApplicationType::SimulationContinuation) => {
25                    Ok(ARINCRecord::RunwaySimulationContinuation(
26                        RunwaySimulationContinuationRecord::parse(input)?,
27                    ))
28                }
29                _ => Err(RecordParseError {
30                    message: "Invalid continuation record application type".to_string(),
31                }),
32            }
33        }
34    }
35}
