1//! # ARINC 424 Definitions - Identifiers
2//! This module contains the identifiers for the ARINC 424 data.
3//! Identifiers are simple alphanumeric string fields. The fields may or may not be constrained to be exactly the specified length always.
4//! Conversely, some fields are minimized as needed in the spec to accomodate record layouts.
5//!
6//! Example is 5.6 - Airport/Heliport Identifier which describes a 4-character code that identifies an airport or heliport.
7

158/// 5.143 Grid MORA
159#[derive(Debug, PartialEq, Eq)]
160pub enum GridMora {
161    Mapped(LengthLimitedIdentifier<3, false>),
162    UnknownOrUnmapped,
163}
164
165impl ParseableField for GridMora {
166    fn from_bytes(bytes: &[u8]) -> Result<Option<Self>, FieldParseError> {
167        if bytes.trim_ascii_end().is_empty() {
168            return Ok(None);
169        }
170        Ok(Some(match bytes {
171            b"UNK" => GridMora::UnknownOrUnmapped,
172            _ => {
173                if let Ok(Some(value)) = LengthLimitedIdentifier::<3, false>::from_bytes(bytes) {
174                    GridMora::Mapped(value)
175                } else {
176                    return Err(FieldParseError::new("Invalid grid MORA".to_string()));
177                }
178            }
179        }))
180    }
181}
