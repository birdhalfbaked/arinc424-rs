8pub(super) struct EnrouteAirwayRecords;
9impl EnrouteAirwayRecords {
10    const CONTINUATION_COLUMN: usize = 39;
11    const CONTINUATION_APPLICATION_COLUMN: usize = 40;
12
13    pub fn parse(input: &[u8]) -> Result<ARINCRecord<'_>, RecordParseError> {
14        if is_primary_record(input, Self::CONTINUATION_COLUMN) {
15            Ok(ARINCRecord::EnrouteAirwayPrimary(
16                EnrouteAirwayPrimaryRecord::parse(input)?,
17            ))
18        } else {
19            if let Ok(Some(application_type)) = ContinuationRecordApplicationType::from_bytes(
20                &input[Self::CONTINUATION_APPLICATION_COLUMN - 1
21                    ..Self::CONTINUATION_APPLICATION_COLUMN],
22            ) {
23                match application_type {
24                    ContinuationRecordApplicationType::StandardContinuation => {
25                        Ok(ARINCRecord::EnrouteAirwayContinuation(
26                            EnrouteAirwayContinuationRecord::parse(input)?,
27                        ))
28                    }
29                    ContinuationRecordApplicationType::FlightPlanningContinuation => {
30                        Ok(ARINCRecord::EnrouteAirwayFlightPlanningContinuation(
31                            EnrouteAirwayFlightPlanningContinuationRecord::parse(input)?,
32                        ))
33                    }
34                    _ => Err(RecordParseError::new(
35                        "Invalid continuation record application type".to_string(),
36                        Some(String::from_utf8_lossy(input).into_owned()),
37                    )),
38                }
39            } else {
40                Ok(ARINCRecord::EnrouteAirwayChangedDataContinuation(
41                    EnrouteAirwayChangedDataContinuationRecord::parse(input)?,
42                ))
43            }
44        }
45    }
46}
